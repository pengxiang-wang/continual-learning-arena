 95    def sanity_check(self) -> None:
 96        r"""Check the sanity of the config dict `self.cfg`."""
 97
 98        # check required config fields
 99        required_config_fields = [
100            "cl_paradigm",
101            "train_tasks",
102            "eval_after_tasks",
103            "global_seed",
104            "cl_dataset",
105            "cl_algorithm",
106            "backbone",
107            "optimizer",
108            "trainer",
109            "metrics",
110            "lightning_loggers",
111            "callbacks",
112            "output_dir",
113            # "hydra" is excluded as it doesn't appear
114            "misc",
115        ]
116        for field in required_config_fields:
117            if not self.cfg.get(field):
118                raise KeyError(
119                    f"Field `{field}` is required in the experiment index config."
120                )
121
122        # check cl_paradigm
123        if self.cfg.cl_paradigm not in ["TIL", "CIL"]:
124            raise ValueError(
125                f"Field cl_paradigm should be either 'TIL' or 'CIL' but got {self.cfg.cl_paradigm}!"
126            )
127
128        if self.cfg.cl_dataset.get("num_tasks"):
129            num_tasks = self.cfg.cl_dataset.get("num_tasks")
130        elif self.cfg.cl_dataset.get("class_split"):
131            num_tasks = len(self.cfg.cl_dataset.class_split)
132        elif self.cfg.cl_dataset.get("datasets"):
133            num_tasks = len(self.cfg.cl_dataset.datasets)
134        else:
135            raise KeyError(
136                "num_tasks is required in cl_dataset config. Please specify `num_tasks` (for `CLPermutedDataset`) or `class_split` (for `CLSplitDataset`) or `datasets` (for `CLCombinedDataset`) in cl_dataset config."
137            )
138
139        # check train_tasks
140        train_tasks = self.cfg.train_tasks
141        if isinstance(train_tasks, ListConfig):
142            if len(train_tasks) < 1:
143                raise ValueError("`train_tasks` must contain at least one task.")
144            if any(t < 1 or t > num_tasks for t in train_tasks):
145                raise ValueError(
146                    f"All task IDs in `train_tasks` must be between 1 and {num_tasks}."
147                )
148        elif isinstance(train_tasks, int):
149            if train_tasks < 0 or train_tasks > num_tasks:
150                raise ValueError(
151                    f"`train_tasks` as integer must be between 0 and {num_tasks}."
152                )
153        else:
154            raise TypeError(
155                "`train_tasks` must be either a list of integers or an integer."
156            )
157
158        # check eval_after_tasks
159        eval_after_tasks = self.cfg.eval_after_tasks
160        if isinstance(eval_after_tasks, ListConfig):
161            if len(eval_after_tasks) < 1:
162                raise ValueError("`eval_after_tasks` must contain at least one task.")
163            if any(t < 1 or t > num_tasks for t in eval_after_tasks):
164                raise ValueError(
165                    f"All task IDs in `eval_after_tasks` must be between 1 and {num_tasks}."
166                )
167        elif isinstance(eval_after_tasks, int):
168            if eval_after_tasks < 0 or eval_after_tasks > num_tasks:
169                raise ValueError(
170                    f"`eval_after_tasks` as integer must be between 0 and {num_tasks}."
171                )
172        else:
173            raise TypeError(
174                "`eval_after_tasks` must be either a list of integers or an integer."
175            )
176
177        # check that eval_after_tasks is a subset of train_tasks
178        if isinstance(train_tasks, list) and isinstance(eval_after_tasks, list):
179            if not set(eval_after_tasks).issubset(set(train_tasks)):
180                raise ValueError(
181                    "`eval_after_tasks` must be a subset of `train_tasks`."
182                )

414    def instantiate_global(self) -> None:
415        r"""Instantiate global components for the entire CL experiment from `self.cfg`."""
416
417        self.instantiate_cl_dataset(self.cfg.cl_dataset)
418        self.instantiate_backbone(self.cfg.backbone)
419        self.instantiate_heads(self.cl_paradigm, self.cfg.backbone.output_dim)

425        r"""Let CL dataset know the CL paradigm to define its CL class map."""

430    def instantiate_task_specific(self) -> None:
431        r"""Instantiate task-specific components for the current task `self.task_id` from `self.cfg`."""
432
433        self.instantiate_optimizer(
434            self.cfg.optimizer,
435            self.task_id,
436        )
437        if self.cfg.get("lr_scheduler"):
438            self.instantiate_lr_scheduler(
439                self.cfg.lr_scheduler,
440                self.task_id,
441            )
442        self.instantiate_callbacks(self.cfg.metrics, self.cfg.callbacks, self.task_id)
443        self.instantiate_lightning_loggers(self.cfg.lightning_loggers, self.task_id)
444        self.instantiate_trainer(
445            self.cfg.trainer, self.task_id
446        )  # trainer should be instantiated after loggers and callbacks
447
448    def setup_task_specific(self) -> None:
449        r"""Setup task-specific components to get ready for the current task `self.task_id`."""
450
451        self.cl_dataset.setup_task_id(self.task_id)
452        self.backbone.setup_task_id(self.task_id)
453        self.model.setup_task_id(
454            self.task_id,
455            len(self.cl_dataset.get_cl_class_map(self.task_id)),
456            self.optimizer_t,
457            lr_scheduler=self.lr_scheduler_t if self.cfg.get("lr_scheduler") else None,
458        )
459
460        pylogger.debug(
461            "Datamodule, model and loggers are all set up ready for task %d!",
462            self.task_id,
463        )
464
465    def run(self) -> None:
466        r"""The main method to run the continual learning experiment."""
467
468        self.instantiate_global()
469        self.setup_global()
470
471        # task loop
472        for task_id in self.train_tasks:
473
474            self.setup_task_id(task_id)
475            self.instantiate_task_specific()
476            self.setup_task_specific()
477
478            # train and validate the model
479            self.trainer_t.fit(
480                model=self.model,
481                datamodule=self.cl_dataset,
482            )
483
484            # evaluation after training and validation
485            if task_id in self.eval_after_tasks:
486                self.trainer_t.test(
487                    model=self.model,
488                    datamodule=self.cl_dataset,
489                )
