 86    def sanity_check(self) -> None:
 87        r"""Check the sanity of the config dict `self.cfg`."""
 88
 89        # check required config fields
 90        required_config_fields = [
 91            "train_tasks",
 92            "eval_tasks",
 93            "global_seed",
 94            "mtl_algorithm",
 95            "backbone",
 96            "optimizer",
 97            "lr_scheduler",
 98            "trainer",
 99            "metrics",
100            "lightning_loggers",
101            "callbacks",
102            "output_dir",
103            # "hydra" is excluded as it doesn't appear
104            "misc",
105        ]
106
107        for field in required_config_fields:
108            if not self.cfg.get(field):
109                raise KeyError(
110                    f"Field `{field}` is required in the experiment index config."
111                )
112
113        if self.cfg.get("mtl_dataset") and self.cfg.get("cl_dataset"):
114            raise KeyError(
115                "Field mtl_dataset and cl_dataset are both specified! Please clarify whether you want to use the specified MTL dataset or construct the MTL dataset from the CL dataset!"
116            )
117        if not self.cfg.get("mtl_dataset") and not self.cfg.get("cl_dataset"):
118            raise KeyError(
119                "Field `mtl_dataset` or `cl_dataset` is required in the experiment index config. "
120            )
121
122        if self.cfg.get("cl_dataset"):
123            if self.cfg.cl_dataset.get("num_tasks"):
124                num_tasks = self.cfg.cl_dataset.get("num_tasks")
125            elif self.cfg.cl_dataset.get("class_split"):
126                num_tasks = len(self.cfg.cl_dataset.class_split)
127            elif self.cfg.cl_dataset.get("datasets"):
128                num_tasks = len(self.cfg.cl_dataset.datasets)
129            else:
130                raise KeyError(
131                    "num_tasks is required in cl_dataset config. Please specify `num_tasks` (for `CLPermutedDataset`) or `class_split` (for `CLSplitDataset`) or `datasets` (for `CLCombinedDataset`) in cl_dataset config."
132                )
133        elif self.cfg.get("mtl_dataset"):
134            if self.cfg.mtl_dataset.get("num_tasks"):
135                num_tasks = self.cfg.mtl_dataset.num_tasks
136            else:
137                raise KeyError(
138                    "num_tasks is required in mtl_dataset config. Please specify `num_tasks` in mtl_dataset config."
139                )
140
141        # check train_tasks
142        train_tasks = self.cfg.train_tasks
143        if isinstance(train_tasks, list):
144            if len(train_tasks) < 1:
145                raise ValueError("`train_tasks` must contain at least one task.")
146            if any(t < 1 or t > num_tasks for t in train_tasks):
147                raise ValueError(
148                    f"All task IDs in `train_tasks` must be between 1 and {num_tasks}."
149                )
150        elif isinstance(train_tasks, int):
151            if train_tasks < 0 or train_tasks > num_tasks:
152                raise ValueError(
153                    f"`train_tasks` as integer must be between 0 and {num_tasks}."
154                )
155        else:
156            raise TypeError(
157                "`train_tasks` must be either a list of integers or an integer."
158            )
159
160        # check eval_tasks
161        eval_tasks = self.cfg.eval_tasks
162        if isinstance(eval_tasks, list):
163            if len(eval_tasks) < 1:
164                raise ValueError("`eval_tasks` must contain at least one task.")
165            if any(t < 1 or t > num_tasks for t in eval_tasks):
166                raise ValueError(
167                    f"All task IDs in `eval_after_tasks` must be between 1 and {num_tasks}."
168                )
169        elif isinstance(eval_tasks, int):
170            if eval_tasks < 0 or eval_tasks > num_tasks:
171                raise ValueError(
172                    f"`eval_tasks` as integer must be between 0 and {num_tasks}."
173                )
174        else:
175            raise TypeError(
176                "`eval_tasks` must be either a list of integers or an integer."
177            )

180        self, mtl_dataset_cfg: DictConfig | None, cl_dataset_cfg: DictConfig | None

193            self.mtl_dataset = hydra.utils.instantiate(
194                mtl_dataset_cfg,

196
197            pylogger.debug(
198                "MTL dataset <%s> (clarena.mtl_datasets.MTLDataset) instantiated! ",
199                mtl_dataset_cfg.get("_target_"),
200            )
201        elif self.cfg.get("cl_dataset"):
202            pylogger.debug(
203                "Constructing MTL dataset (clarena.mtl_datasets.MTLDataset) from <%s> (clarena.cl_datasets.CLDataset) ...",
204                cl_dataset_cfg.get("_target_"),
205            )
206
207            self.cl_dataset = hydra.utils.instantiate(
208                cl_dataset_cfg
209            )  # instantiate the CL dataset
210            self.mtl_dataset = MTLDatasetFromCL(
211                self.cl_dataset,
212            )  # construct the MTL dataset from the CL dataset
213            pylogger.debug(
214                "MTL dataset (clarena.mtl_datasets.MTLDataset) constructed from <%s> (clarena.cl_datasets.CLDataset)! ",
215                cl_dataset_cfg.get("_target_"),
216            )

383    def instantiate(self) -> None:
384        r"""Instantiate components for the MTL experiment from `self.cfg`."""
385
386        if self.cfg.get("cl_dataset"):
387            self.instantiate_mtl_dataset(
388                cl_dataset_cfg=self.cfg.cl_dataset, mtl_dataset_cfg=None
389            )
390        elif self.cfg.get("mtl_dataset"):
391            self.instantiate_mtl_dataset(
392                mtl_dataset_cfg=self.cfg.mtl_dataset, cl_dataset_cfg=None
393            )
394        self.instantiate_backbone(self.cfg.backbone)
395        self.instantiate_heads(
396            input_dim=self.cfg.backbone.output_dim,
397        )
398        self.instantiate_mtl_algorithm(
399            self.cfg.mtl_algorithm
400        )  # mtl_algorithm should be instantiated after backbone and heads
401        self.instantiate_optimizer(self.cfg.optimizer)
402        self.instantiate_lr_scheduler(self.cfg.lr_scheduler)
403        self.instantiate_callbacks(self.cfg.metrics, self.cfg.callbacks)
404        self.instantiate_lightning_loggers(self.cfg.lightning_loggers)
405        self.instantiate_trainer(
406            self.cfg.trainer
407        )  # trainer should be instantiated after loggers and callbacks
408
409    def setup(self) -> None:
410        r"""Setup."""
411        self.set_global_seed()
412
413        # set up the CL dataset if exists
414        if self.cfg.get("cl_dataset"):
415            self.cl_dataset.set_cl_paradigm(
416                cl_paradigm="TIL"
417            )  # MTL requires independent heads
418            for task_id in self.train_tasks:
419                self.cl_dataset.setup_task_id(task_id)
420
421        self.mtl_dataset.setup_tasks(
422            train_tasks=self.train_tasks,
423            eval_tasks=self.eval_tasks,
424        )
425        self.heads.setup_tasks(
426            task_ids=self.train_tasks,
427            num_classes={
428                task_id: len(self.mtl_dataset.get_class_map(task_id))
429                for task_id in self.train_tasks
430            },
431        )
432        self.model.setup_tasks(self.optimizer, self.lr_scheduler)
433
434    def run(self) -> None:
435        r"""The main method to run the multi-task learning experiment."""
436
437        self.instantiate()
438        self.setup()
439
440        # fit the model on the MTL dataset
441        self.trainer.fit(
442            model=self.model,
443            datamodule=self.mtl_dataset,
444        )
445
446        # evaluation after training and validation
447        self.trainer.test(
448            model=self.model,
449            datamodule=self.mtl_dataset,
450        )
