 47        r"""Initialize the FaceScrub10 dataset.
 48
 49        **Args:**
 50        - **root** (`str` | `Path`): Root directory of the dataset.

 53        - **target_transform** (`Callable`, optional): Transform to apply to each label.

 69    def _check_integrity(self) -> bool:
 70        """Check if dataset files exist."""
 71        file_path = os.path.join(
 72            self.root, self.train_file if self.train else self.test_file
 73        )
 74        return os.path.exists(file_path)
 75
 76    def _download(self) -> None:
 77        """Download and extract the dataset."""
 78        os.makedirs(self.root, exist_ok=True)
 79        archive_path = os.path.join(self.root, self.filename)
 80
 81        if not os.path.exists(archive_path):
 82            pylogger.debug(f"Downloading from {self.url}")
 83            urllib.request.urlretrieve(self.url, archive_path)
 84
 85        with zipfile.ZipFile(archive_path, "r") as zip_ref:
 86            zip_ref.extractall(self.root)
 87        pylogger.debug("Downloaded and extracted dataset.")
 88
 89    def _load_data(self) -> None:
 90        """Load features and labels from pickle."""
 91        file_path = os.path.join(

 94
 95        with open(file_path, "rb") as f:
