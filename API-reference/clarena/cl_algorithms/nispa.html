 55    def on_train_start(self) -> None:
 56        """Initialize all masks at the very beginning of Task 1."""
 57        self.best_phase_acc = 0.0
 58        self.phase_idx = 0
 59
 60        if self.task_id == 1:
 61            # zero‐out the backbone’s parameter masks
 62            self.backbone.initialize_parameter_mask()
 63
 64            for layer in self.backbone.weighted_layer_names:
 65                # number of units = output dim of that layer
 66                num_units = self.backbone.get_layer_by_name(layer).weight.shape[0]
 67                self.candidate_stable_unit_mask_t[layer] = torch.zeros(

 72                )
 73                self.plastic_unit_mask_t[layer] = torch.ones(
 74                    num_units, device=self.device
 75                )

 77    def clip_grad_by_frozen_mask(self) -> None:
 78        """Zero‐out grads on frozen connections."""
 79        for layer_name in self.backbone.weighted_layer_names:
 80            layer = self.backbone.get_layer_by_name(layer_name)
 81            layer.weight.grad.data *= (
 82                1.0 - self.backbone.frozen_weight_mask_t[layer_name]
 83            )
 84            if layer.bias is not None:
 85                layer.bias.grad.data *= (
 86                    1.0 - self.backbone.frozen_bias_mask_t[layer_name]
 87                )

 90        self,
 91        input: torch.Tensor,
 92        stage: str,
 93        task_id: int | None = None,
 94    ) -> tuple[Tensor, Dict[str, Tensor], Dict[str, Tensor], Dict[str, Tensor]]:
 95        """Wrapper around the backbone’s forward + the heads."""

100    def training_step(self, batch: Any) -> dict[str, Tensor]:
101        x, y = batch
102        opt = self.optimizers()
103        opt.zero_grad()
104
105        logits, w_mask, b_mask, _ = self.forward(x, stage="train", task_id=self.task_id)
106        loss = self.criterion(logits, y)
107        self.manual_backward(loss)
108        self.clip_grad_by_frozen_mask()
109        opt.step()
110
111        acc = (logits.argmax(dim=1) == y).float().mean()
112        return {"loss": loss, "acc": acc}

114    def on_validation_epoch_end(self, outputs: dict[str, Any]) -> None:
115        """Every num_epochs_per_phase epochs, do a phase‐end check, selection, and rewire."""
116        if (self.current_epoch + 1) % self.num_epochs_per_phase != 0:
117            return
118
119        # cache model + optimizer in case we need to rollback
120        cached = {
121            "model": deepcopy(self.state_dict()),
122            "opt": deepcopy(self.trainer.optimizers[0].state_dict()),
123        }
124
125        val_acc = outputs["acc"].item()
126        self.best_phase_acc = max(self.best_phase_acc, val_acc)
127
128        if val_acc >= self.best_phase_acc - self.accuracy_fall_threshold:
129            # still “good enough” → select + rewire
130            tau = 0.5 * (1 + math.cos(self.phase_idx * math.pi / self.k))
131            self.select_candidate_stable_units(activation_fraction=tau)
132
133            # merge candidate → stable, update plastic = complement
134            for layer in self.backbone.weighted_layer_names:
135                self.stable_unit_mask_t[layer] = (
136                    self.stable_unit_mask_t[layer]
137                    | self.candidate_stable_unit_mask_t[layer]
138                ).float()
139                self.plastic_unit_mask_t[layer] = 1.0 - self.stable_unit_mask_t[layer]
140
141            # update the backbone’s parameter‐level masks from these unit‐masks
142            self._update_parameter_masks()
143
144            # drop & grow
145            dropped = self.drop_connections_plastic_to_stable()
146            self.grow_new_connections(dropped)
147
148            self.phase_idx += 1
149
150        else:
151            # roll back to previous checkpoint
152            self.load_state_dict(cached["model"])
153            self.trainer.optimizers[0].load_state_dict(cached["opt"])
154            pylogger.info(f"Phase {self.phase_idx} terminated early (val_acc fell).")
155
156    def _update_parameter_masks(self) -> None:
157        """Convert each layer’s unit‐mask → a weight/bias mask."""
158        for layer in self.backbone.weighted_layer_names:
159            # gather shapes
160            W = self.backbone.weight_mask_t[layer]
161            shape = W.shape
162            stable_units = self.stable_unit_mask_t[layer].bool()
163            plastic_units = self.plastic_unit_mask_t[layer].bool()
164
165            # out-unit plastic mask
166            out_pl = plastic_units.view(-1, 1).expand(shape)
167
168            # in-unit plastic mask (preceding layer)
169            prev = self.backbone.preceding_layer_name(layer)
170            if prev:
171                in_pl = self.plastic_unit_mask_t[prev].view(1, -1).expand(shape)
172            else:
173                in_pl = torch.ones(shape, device=self.device, dtype=torch.bool)
174
175            # new weight mask = plastic→plastic
176            self.backbone.weight_mask_t[layer] = (out_pl & in_pl).float()
177            # bias mask = plastic units only
178            self.backbone.bias_mask_t[layer] = plastic_units.float()
179
180    def select_candidate_stable_units(self, activation_fraction: float) -> None:
181        """Pick the top‐activated units until we reach `activation_fraction` of total."""
182        # accumulate per‐unit activations over the training set
183        summed = {
184            layer: torch.zeros_like(self.stable_unit_mask_t[layer])
185            for layer in self.backbone.weighted_layer_names
186        }
187        total_data = 0
188
189        for x, _ in self.trainer.datamodule.train_dataloader():
190            x = x.to(self.device)
191            _, _, _, acts = self.forward(x, stage="validation")
192            batch = x.shape[0]
193            total_data += batch
194
195            for layer in summed:
196                # sum up the ∥activation∥ per unit
197                summed[layer] += acts[layer].sum(dim=0)  # sum over batch
198
199        # threshold per‐layer
200        for layer in summed:
201            vals = summed[layer]
202            cutoff = vals.sum() * activation_fraction
203            cum = 0.0
204            mask = torch.zeros_like(vals)
205
206            # pick largest values until cum ≥ cutoff
207            while cum < cutoff:
208                idx = torch.argmax(vals)
209                cum += vals[idx].item()
210                mask[idx] = 1.0
211                vals[idx] = 0.0  # ensure we don’t pick it twice
212
213            self.candidate_stable_unit_mask_t[layer] = mask.to(self.device)

215    def drop_connections_plastic_to_stable(self) -> Dict[str, int]:
216        """
217        Drop connections *from* plastic units *into* stable units:
218        i.e. weight[stable_out, plastic_in] → 0, bias[stable_out] → 0.
219        """
220        dropped_counts: Dict[str, int] = {}
221
222        for layer in self.backbone.weighted_layer_names:
223            Wm = self.backbone.weight_mask_t[layer]
224            bm = self.backbone.bias_mask_t[layer]
225
226            stable_out = self.stable_unit_mask_t[layer].bool()
227            prev = self.backbone.preceding_layer_name(layer)
228            if prev:
229                plastic_in = (~self.stable_unit_mask_t[prev]).bool()
230            else:
231                plastic_in = torch.zeros(
232                    Wm.shape[1], device=self.device, dtype=torch.bool
233                )
234
235            before_w = int(Wm.sum().item())
236            # zero out those connections
237            Wm[stable_out, :][:, plastic_in] = 0
238            after_w = int(Wm.sum().item())
239
240            before_b = int(bm.sum().item())
241            bm[stable_out] = 0
242            after_b = int(bm.sum().item())
243
244            self.backbone.weight_mask_t[layer] = Wm
245            self.backbone.bias_mask_t[layer] = bm
246            dropped_counts[layer] = (before_w - after_w) + (before_b - after_b)
247
248        return dropped_counts
249
250    def grow_new_connections(self, dropped: Dict[str, int]) -> None:
251        """
252        Regrow new connections *among plastic units* to keep the network at a fixed density.
253        For simplicity, we randomly reassign the dropped number of connections.
254        """
255        for layer, num_drop in dropped.items():
256            if num_drop <= 0:
257                continue
258
259            Wm = self.backbone.weight_mask_t[layer]
260            prev = self.backbone.preceding_layer_name(layer)
261
262            stable_out = self.stable_unit_mask_t[layer].bool()
263            plastic_out = ~stable_out
264            if prev:
265                plastic_in = (~self.stable_unit_mask_t[prev]).bool()
266            else:
267                plastic_in = torch.ones(
268                    Wm.shape[1], device=self.device, dtype=torch.bool
269                )
270
271            # find all available plastic→plastic positions that are currently zero
272            mask = (Wm == 0) & plastic_out.view(-1, 1) & plastic_in.view(1, -1)
273            coords = torch.nonzero(mask, as_tuple=False)
274            if coords.numel() == 0:
275                continue
276
277            coords = coords.tolist()
278            chosen = random.sample(coords, min(len(coords), num_drop))
279            for i, j in chosen:
280                Wm[i, j] = 1.0
281
282            # for bias, simply ensure all plastic units have bias = 1
283            bm = self.backbone.bias_mask_t[layer]
284            bm[plastic_out] = 1.0
285
286            self.backbone.weight_mask_t[layer] = Wm
287            self.backbone.bias_mask_t[layer] = bm
288
289    def on_train_end(self) -> None:
290        """
291        Called at the very end of a task:
292        freeze *all* connections that were used (mask = 1) as part of this task.
293        """
294        for layer in self.backbone.weighted_layer_names:
295            # freeze any connection that was ever allowed (weight_mask_t == 1)
296            self.backbone.frozen_weight_mask_t[layer] |= (
297                self.backbone.weight_mask_t[layer] > 0
298            ).float()
299            self.backbone.frozen_bias_mask_t[layer] |= (
300                self.backbone.bias_mask_t[layer] > 0
301            ).float()

306        loss = self.criterion(logits, y)
307        acc = (logits.argmax(dim=1) == y).float().mean()
308        return {"loss": loss, "acc": acc}

115    def on_validation_epoch_end(self, outputs: dict[str, Any]) -> None:
116        """Every num_epochs_per_phase epochs, do a phase‐end check, selection, and rewire."""
117        if (self.current_epoch + 1) % self.num_epochs_per_phase != 0:
118            return
119
120        # cache model + optimizer in case we need to rollback
121        cached = {
122            "model": deepcopy(self.state_dict()),
123            "opt": deepcopy(self.trainer.optimizers[0].state_dict()),
124        }
125
126        val_acc = outputs["acc"].item()
127        self.best_phase_acc = max(self.best_phase_acc, val_acc)
128
129        if val_acc >= self.best_phase_acc - self.accuracy_fall_threshold:
130            # still “good enough” → select + rewire
131            tau = 0.5 * (1 + math.cos(self.phase_idx * math.pi / self.k))

181    def select_candidate_stable_units(self, activation_fraction: float) -> None:
182        """Pick the top‐activated units until we reach `activation_fraction` of total."""
183        # accumulate per‐unit activations over the training set
184        summed = {
185            layer: torch.zeros_like(self.stable_unit_mask_t[layer])
186            for layer in self.backbone.weighted_layer_names
187        }
188        total_data = 0
189
190        for x, _ in self.trainer.datamodule.train_dataloader():
191            x = x.to(self.device)
192            _, _, _, acts = self.forward(x, stage="validation")
193            batch = x.shape[0]
194            total_data += batch
195
196            for layer in summed:
197                # sum up the ∥activation∥ per unit
198                summed[layer] += acts[layer].sum(dim=0)  # sum over batch
199
200        # threshold per‐layer
201        for layer in summed:
202            vals = summed[layer]
203            cutoff = vals.sum() * activation_fraction

206
207            # pick largest values until cum ≥ cutoff

212                vals[idx] = 0.0  # ensure we don’t pick it twice

def drop_connections_plastic_to_stable(self) -> Dict[str, int]: 
216    def drop_connections_plastic_to_stable(self) -> Dict[str, int]:
217        """
218        Drop connections *from* plastic units *into* stable units:
219        i.e. weight[stable_out, plastic_in] → 0, bias[stable_out] → 0.
220        """
221        dropped_counts: Dict[str, int] = {}
222
223        for layer in self.backbone.weighted_layer_names:
224            Wm = self.backbone.weight_mask_t[layer]
225            bm = self.backbone.bias_mask_t[layer]
226
227            stable_out = self.stable_unit_mask_t[layer].bool()
228            prev = self.backbone.preceding_layer_name(layer)
229            if prev:
230                plastic_in = (~self.stable_unit_mask_t[prev]).bool()
231            else:
232                plastic_in = torch.zeros(
233                    Wm.shape[1], device=self.device, dtype=torch.bool
234                )
235
236            before_w = int(Wm.sum().item())
237            # zero out those connections
238            Wm[stable_out, :][:, plastic_in] = 0
239            after_w = int(Wm.sum().item())
240
241            before_b = int(bm.sum().item())
242            bm[stable_out] = 0
243            after_b = int(bm.sum().item())
244
245            self.backbone.weight_mask_t[layer] = Wm
246            self.backbone.bias_mask_t[layer] = bm
247            dropped_counts[layer] = (before_w - after_w) + (before_b - after_b)
248
249        return dropped_counts

252        """
253        Regrow new connections *among plastic units* to keep the network at a fixed density.
254        For simplicity, we randomly reassign the dropped number of connections.
255        """
256        for layer, num_drop in dropped.items():
257            if num_drop <= 0:
258                continue
259
260            Wm = self.backbone.weight_mask_t[layer]
261            prev = self.backbone.preceding_layer_name(layer)
262
263            stable_out = self.stable_unit_mask_t[layer].bool()
264            plastic_out = ~stable_out
265            if prev:
266                plastic_in = (~self.stable_unit_mask_t[prev]).bool()
267            else:
268                plastic_in = torch.ones(
269                    Wm.shape[1], device=self.device, dtype=torch.bool
270                )
271
272            # find all available plastic→plastic positions that are currently zero
273            mask = (Wm == 0) & plastic_out.view(-1, 1) & plastic_in.view(1, -1)

275            if coords.numel() == 0:
276                continue
