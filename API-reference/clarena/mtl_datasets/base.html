224    def train_and_val_transforms(self, task_id: int) -> transforms.Compose:
225        r"""Transforms generator for train and validation dataset of task `task_id` incorporating the custom transforms with basic transforms like `normalization` and `ToTensor()`. It is a handy tool to use in subclasses when constructing the dataset.
226
227        **Args:**
228        - **task_id** (`int`): the task ID of training and validation dataset to get the transforms for.
229
230        **Returns:**
231        - **train_and_val_transforms** (`transforms.Compose`): the composed training transforms.
232        """
233        repeat_channels_transform = (
234            transforms.Grayscale(num_output_channels=self.repeat_channels[task_id])
235            if self.repeat_channels[task_id] is not None
236            else None
237        )
238        to_tensor_transform = transforms.ToTensor() if self.to_tensor[task_id] else None
239        resize_transform = (
240            transforms.Resize(self.resize[task_id])
241            if self.resize[task_id] is not None
242            else None
243        )
244        normalization_transform = transforms.Normalize(
245            self.mean[task_id], self.std[task_id]
246        )
247
248        return transforms.Compose(
249            list(
250                filter(
251                    None,
252                    [
253                        repeat_channels_transform,
254                        to_tensor_transform,
255                        resize_transform,
256                        self.custom_transforms[task_id],
257                        normalization_transform,
258                    ],
259                )
260            )
261        )  # the order of transforms matters
